Hv-Autocomplete

HV-Autocomplete is the most flexible autoсomplete for design and developers. You can do something more then with another autoсomplets. For example firstly goal HV-Autocomplete is you may do Horizontal or Vertical search results.

HV-Autocomplete is using jQuery.

Styling

Class

JS

  var configs = {
    div: $(".my-autocomplete-class"),
    inputPlaceholder: "Search...",
    categories: true,
    inputClassName: "example-class-input",
    resultClassName: "example-class-result"
  };

  HVAutocomplite(configs, data);

CSS

  .example-class-input {
    width: 300px;
  }
  
  .example-class-result {
    width: 300px;
    background-color: #ff8742;
    color: #fff;
  }
  
  .example-class-result p a {
    color: #fff;
  }

Style

JS

  var configs = {
    div: $(".my-autocomplete-horizontal"),
    inputPlaceholder: "Search...",
    inputStyles: {
      paddingLeft: "30px",  
      width: 300
    },
    resultStyles: {
      backgroundColor: "#efefef",
      width: 300
    }
  };
  
  HVAutocomplite(configs, data);

Categories

JS

var configs = {
  div: $(".my-autocomplete-horizontal"),
  inputPlaceholder: "Search...",
  categories: true,
  display: "horizontal"
};

HVAutocomplite(configs, data);
